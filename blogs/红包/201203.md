<template>
  <div class="box">
     <div class="lottie" ref="redPacket"></div>
     <div class="money-lottie" ref="mony"></div>
     <div class="money2-lottie" ref="mony2"></div>
     <div class="ribbon-lottie" ref="ribbon"></div>
  </div>
</template>
<script>
import lottie from 'lottie-web';
import RedPcket from '~public/example/lottie/red-packet.json'
import Mony from '~public/example/lottie/money.json'
import Mony2 from '~public/example/lottie/money2.json'
import Ribbon from '~public/example/lottie/ribbon.json'
export default {
  data () {
    return {}
  },
  mounted () {
    const redPcketParams = {
      container: this.$refs.redPacket,
      renderer: "svg",
      loop: true,
      autoplay: true,
      animationData: RedPcket
    };

    const monyParams = {
      container: this.$refs.mony,
      renderer: "svg",
      loop: true,
      autoplay: true,
      animationData: Mony
    };

    const mony2Params = {
      container: this.$refs.mony2,
      renderer: "svg",
      loop: true,
      autoplay: true,
      animationData: Mony2
    };

    const ribbonParams = {
      container: this.$refs.ribbon,
      renderer: "svg",
      loop: true,
      autoplay: true,
      animationData: Ribbon
    };

    lottie.loadAnimation(ribbonParams);
    lottie.loadAnimation(mony2Params);
    lottie.loadAnimation(redPcketParams);
    lottie.loadAnimation(monyParams);
  },
  methods: {
    
  }
}
</script>
